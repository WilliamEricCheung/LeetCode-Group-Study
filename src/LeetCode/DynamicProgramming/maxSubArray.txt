class Solution:
    def maxSubArray(self, nums) -> int:
        if not nums:
            return 0
        final = nums[0]
        temp = nums[0]
        if len(nums) > 1:
            for i in range(1, len(nums)):
                temp = max(temp + nums[i], nums[i])
                if temp >= final:
                    final = temp
        return final
