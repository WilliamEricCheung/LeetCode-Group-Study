class ListNode:
    def __init__(self,x):
        self.val = x
        self.next = None

class Solution:
    def mergetwolists(self, l1: ListNode, l2: ListNode) -> ListNode:
        newlist = ListNode(None)
        cur = newlist
        while l1 or l2:
            if not l1:
                cur.next = l2
                break
            if not l2:
                cur.next = l1
                break
            if l1 and l2:
                if l1.val <= l2.val:
                    cur.next = l1
                    cur = cur.next
                    l1 = l1.next
                else:
                    cur.next = l2
                    cur = cur.next
                    l2 = l2.next
        return cur.next
